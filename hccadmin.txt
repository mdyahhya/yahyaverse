    // Convert VAPID key
    function urlBase64ToUint8Array(base64String) {
        const padding = '='.repeat((4 - base64String.length % 4) % 4);
        const base64 = (base64String + padding)
            .replace(/\-/g, '+')
            .replace(/_/g, '/');

        const rawData = window.atob(base64);
        const outputArray = new Uint8Array(rawData.length);

        for (let i = 0; i < rawData.length; ++i) {
            outputArray[i] = rawData.charCodeAt(i);
        }
        return outputArray;
    }

// Enable Admin Notifications
async function enableAdminNotifications() {
    const btn = document.getElementById('notificationBtn');
    
    if (!('Notification' in window)) {
        alert('‚ùå Notifications not supported in your browser');
        return;
    }

    if (!('serviceWorker' in navigator) || !('PushManager' in window)) {
        alert('‚ùå Push notifications not supported in your browser');
        return;
    }

    // Check if already enabled
    const currentPermission = Notification.permission;
    if (currentPermission === 'granted') {
        const isSubscribed = localStorage.getItem('admin_notifications_enabled') === 'true';
        if (isSubscribed) {
            alert('‚úÖ Notifications are already enabled!');
            return;
        }
    }

    try {
        btn.disabled = true;
        btn.innerHTML = '‚è≥ Enabling...';

        const permission = await Notification.requestPermission();
        
        if (permission !== 'granted') {
            alert('‚ùå Notification permission denied.\n\nTo enable:\n1. Click the lock icon in address bar\n2. Allow notifications\n3. Click the button again');
            btn.disabled = false;
            btn.innerHTML = 'üîî Enable Alerts';
            return;
        }

        // Subscribe to push
        const registration = await navigator.serviceWorker.ready;
        
        let subscription = await registration.pushManager.getSubscription();

        if (!subscription) {
            subscription = await registration.pushManager.subscribe({
                userVisibleOnly: true,
                applicationServerKey: urlBase64ToUint8Array(VAPID_PUBLIC_KEY)
            });
        }

        // Save to database
        const subscriptionJson = subscription.toJSON();

        const { error } = await supabaseClient
            .from('push_subscriptions')
            .upsert({
                endpoint: subscription.endpoint,
                p256dh: subscriptionJson.keys.p256dh,
                auth: subscriptionJson.keys.auth,
                user_agent: 'Admin - ' + navigator.userAgent,
                device_type: 'admin',
                is_active: true,
                last_used: new Date().toISOString()
            }, {
                onConflict: 'endpoint'
            });

        if (error) throw error;

        localStorage.setItem('admin_notifications_enabled', 'true');
        
        btn.innerHTML = '‚úÖ Alerts Enabled';
        btn.classList.add('disabled');
        btn.disabled = true;
        btn.style.background = 'var(--neutral-warm)';

        alert('‚úÖ Success!\n\nYou will now receive notifications for:\n‚Ä¢ New admission inquiries\n‚Ä¢ Important updates');

    } catch (error) {
        console.error('‚ùå Failed to enable notifications:', error);
        alert('‚ùå Failed to enable notifications.\n\nError: ' + error.message);
        btn.disabled = false;
        btn.innerHTML = 'üîî Enable Alerts';
    }
}

    } catch (error) {
        console.error('Install error:', error);
        installBtn.disabled = false;

    }
}

// TOGGLE NOTIFICATION FUNCTION
// ============================================
async function toggleNotifications() {
    const isEnabled = localStorage.getItem('admin_notifications_enabled') === 'true' || 
                      Notification.permission === 'granted';

    if (isEnabled) {
        if (confirm('Disable notification alerts for new inquiries?')) {
            await unsubscribeAdminNotifications();
        }
    } else {
        await subscribeAdminToNotifications();
    }
}

// Upload image to Supabase Storage
async function uploadImageToStorage(file, folder = 'general') {
    try {
        // Validate file
        if (!file || !file.type.startsWith('image/')) {
            throw new Error('Please select a valid image file');
        }

        // Check file size (max 5MB)
        const maxSize = 5 * 1024 * 1024; // 5MB
        if (file.size > maxSize) {
            throw new Error('Image size must be less than 5MB');
        }

        // Generate unique filename
        const fileExt = file.name.split('.').pop();
        const fileName = `${folder}/${Date.now()}-${Math.random().toString(36).substring(7)}.${fileExt}`;

        console.log('üì§ Uploading image:', fileName);

        // Upload to Supabase Storage
        const { data, error } = await supabaseClient.storage
            .from('hcc-uploads')
            .upload(fileName, file, {
                cacheControl: '3600',
                upsert: false
            });

        if (error) throw error;

        // Get public URL
        const { data: urlData } = supabaseClient.storage
            .from('hcc-uploads')
            .getPublicUrl(fileName);

        console.log('‚úÖ Image uploaded:', urlData.publicUrl);
        return urlData.publicUrl;

    } catch (error) {
        console.error('‚ùå Upload failed:', error);
        throw error;
    }
}

// Delete image from storage
async function deleteImageFromStorage(imageUrl) {
    try {
        if (!imageUrl || !imageUrl.includes('hcc-uploads')) {
            return; // Not from our storage
        }

        // Extract file path from URL
        const urlParts = imageUrl.split('/hcc-uploads/');
        if (urlParts.length < 2) return;

        const filePath = urlParts[1];

        const { error } = await supabaseClient.storage
            .from('hcc-uploads')
            .remove([filePath]);

        if (error) throw error;
        console.log('‚úÖ Image deleted from storage');

    } catch (error) {
        console.error('‚ùå Failed to delete image:', error);
    }
}

// Compress image before upload (optional but recommended)
async function compressImage(file, maxWidth = 1200, quality = 0.8) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.readAsDataURL(file);
        reader.onload = (e) => {
            const img = new Image();
            img.src = e.target.result;
            img.onload = () => {
                const canvas = document.createElement('canvas');
                let width = img.width;
                let height = img.height;

                // Resize if needed
                if (width > maxWidth) {
                    height = (height * maxWidth) / width;
                    width = maxWidth;
                }

                canvas.width = width;
                canvas.height = height;

                const ctx = canvas.getContext('2d');
                ctx.drawImage(img, 0, 0, width, height);

                canvas.toBlob(
                    (blob) => {
                        resolve(new File([blob], file.name, {
                            type: 'image/jpeg',
                            lastModified: Date.now()
                        }));
                    },
                    'image/jpeg',
                    quality
                );
            };
            img.onerror = reject;
        };
        reader.onerror = reject;
    });
}

// Show image preview
function showImagePreview(file, previewElementId) {
    const reader = new FileReader();
    reader.onload = (e) => {
        const preview = document.getElementById(previewElementId);
        if (preview) {
            preview.src = e.target.result;
            preview.style.display = 'block';
        }
    };
    reader.readAsDataURL(file);
}
